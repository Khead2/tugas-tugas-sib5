//masuk pada database tokosib5
USE tokosib5;

//membuat table pembelian yang references dengan table produk
CREATE TABLE pembelian (
    -> tanggal date NOT NULL,
    -> nomor int PRIMARY KEY AUTO_INCREMENT,
    -> produk_id int NOT NULL REFERENCES produk(id),
    -> jumlah int NOT NULL,
    -> vendor_id int NOT NULL);

//melihat descripsi table pembelian
desc pembelian;

//membuat table vendor yang references dengan table pembelian
CREATE TABLE vendor (
    -> nomor int PRIMARY KEY NOT NULL REFERENCES pembelian(vendor_id),
    -> nama varchar(30) NOT NULL,
    -> kota varchar(30) NOT NULL,
    -> kontak int NOT NULL);

//melihat descripsi table vendor
DESC vendor;
